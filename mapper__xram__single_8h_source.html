19 // Macro to declare a variable in XRAM.
20 // - index = XRAM bank index
21 // - declaration = variable declaration
22 // This will declare two variables, one read-only and one write-only.
23 // For example, DECLARE_XRAM_VARIABLE(0, int my_var)
24 // creates two variables: my_var_read and my_var_write.
25 // NOTE: These variables are not initialized by default.
26 #define DECLARE_XRAM_VARIABLE(index, declaration) \
27  __attribute__((section(".xram" #index "_read"))) volatile const declaration##_read; \
28  __attribute__((section(".xram" #index "_write"))) volatile declaration##_write;
